คลาสที่ใช้จัดการข้อมูลที่มีรูปแบบปฏิทิน

คลาส Calendar สร้างมาเพื่อจัดการๆ สร้างข้อมูลในรูปแบบปฏิทิน. ถูกออกแบบมาให้เรา สามารถจัดการทุกส่วนได้อย่างง่ายๆ

การเรียกใช้งาน

จะเรียกเหมือนๆ กับคลาสอื่นๆของ CodeIgniter, โดยใช้เมทอด $this->load->library function:

$this->load->library('calendar');

เมื่อโหลดเเล้ว เราก็สามารถเรียกใช้ได้ ตามนี้: $this->calendar

การแสดงปฏิทิน

ตัวอย่างง่ายๆ ตามนี้ครับ:

$this->load->library('calendar');

echo $this->calendar->generate();

โดยตัวคลาสจะอ้างอิงวันเวลาของ server ถ้าเราต้องการกำหนดค่าก็สามารถใส่ ได้ตามตัวอย่าง:

$this->load->library('calendar');

echo $this->calendar->generate(2006, 6);

ตามตัวอย่าง เราเรียกใช้ค่าของปี 2006 เดือน มิถุนา.

วิธีเขียนข้อมูลลงในช่องวัน

ตามตัวอย่าง เราใช้อาเรย์ที่มีคีย์ตามวันที่ และมีค่าคือข้อมูลที่เราจะป้อนลงไป:

$this->load->library('calendar');

$data = array(
               3  => 'http://example.com/news/article/2006/03/',
               7  => 'http://example.com/news/article/2006/07/',
               13 => 'http://example.com/news/article/2006/13/',
               26 => 'http://example.com/news/article/2006/26/'
             );

echo $this->calendar->generate(2006, 6, $data);

ค่าในช่องของวันที่ 3, 7, 13, และ 26 จะมีลิ้งที่เราใส่ไป.

หมายเหตุ: โดยค่าเริ่มต้นแล้ว ค่าที่ส่งเข้ามาจะมีค่าเป็นลิ้ง ในส่วนต่อไปเราจะมาดูวิธีการตั้งค่าข้อมูลในรุปแบบอื่นๆ .

การตั้งค่าในส่วนการแสดงผล

มีอยู่ 7 รูปแบบด้วยกันนะครับ :

$prefs = array (
               'start_day'    => 'saturday',
               'month_type'   => 'long',
               'day_type'     => 'short'
             );

$this->load->library('calendar', $prefs);

echo $this->calendar->generate();

ตามตัวอย่างเราจะให้ปฏิทินเริ่ม ในวันเสาร์ และใช้ชื่อเดือนในแบบเต็ม ส่วนชื่อวันจะเป็นแบบย่อ.

ชื่อ ค่าเริ่มต้น การตั้งค่าเพิ่มเติม คำอธิบาย
templateไม่มีไม่มี ชื่อของรูปแบบ.
local_timetime() ไม่มีค่า Unix timestamp
start_daysunday ทุกๆวัน (sunday, monday, tuesday, etc.) ใช้กำหนดวันเริ่มต้นของเดือน.
month_typelonglong, short ใช้กำหนดรูปแบบชื่อของเดือน long = January, short = Jan.
day_typeabrlong, short, abr ใช้กำหนดรูปแบบของชื่อวัน. long = Sunday, short = Sun, abr = Su.
show_next_prevFALSETRUE/FALSE (boolean) ใช้กำหนดว่าจะมีลิ้งที่มีเดือนต่อไปไหม.
next_prev_urlNoneA URLใช้กำหนดลิ้งที่จะเลือกไปเดือนก่อนหน้าหรือด้านหลัง.

การแสดงลิ้ง Next/Previous

ในตัวอย่างเรากำหนด url ที่เราจะนำไปแสดงผลในหน้าถัดไป:

$prefs = array (
               'show_next_prev'  => TRUE,
               'next_prev_url'   => 'http://example.com/index.php/calendar/show/'
             );

$this->load->library('calendar', $prefs);

echo $this->calendar->generate($this->uri->segment(3), $this->uri->segment(4));

โดยการตั้งค่าทำได้ดังนี้:

  • เราต้องตั้งค่า "show_next_prev" ให้เป็น TRUE.
  • ในคีย์ "next_prev_url" URL เรากำหนดเป็นเส้นทางที่อยู่ของ controller .

    • ตัวอย่างการสร้าง template

    $prefs['template'] = '

       {table_open}<table border="0" cellpadding="0" cellspacing="0">{/table_open}

       {heading_row_start}<tr>{/heading_row_start}

       {heading_previous_cell}<th><a href="{previous_url}">&lt;&lt;</a></th>{/heading_previous_cell}
       {heading_title_cell}<th colspan="{colspan}">{heading}</th>{/heading_title_cell}
       {heading_next_cell}<th><a href="{next_url}">&gt;&gt;</a></th>{/heading_next_cell}

       {heading_row_end}</tr>{/heading_row_end}

       {week_row_start}<tr>{/week_row_start}
       {week_day_cell}<td>{week_day}</td>{/week_day_cell}
       {week_row_end}</tr>{/week_row_end}

       {cal_row_start}<tr>{/cal_row_start}
       {cal_cell_start}<td>{/cal_cell_start}

       {cal_cell_content}<a href="{content}">{day}</a>{/cal_cell_content}
       {cal_cell_content_today}<div class="highlight"><a href="{content}">{day}</a></div>{/cal_cell_content_today}

       {cal_cell_no_content}{day}{/cal_cell_no_content}
       {cal_cell_no_content_today}<div class="highlight">{day}</div>{/cal_cell_no_content_today}

       {cal_cell_blank}&nbsp;{/cal_cell_blank}

       {cal_cell_end}</td>{/cal_cell_end}
       {cal_row_end}</tr>{/cal_row_end}

       {table_close}</table>{/table_close}
    ';

    $this->load->library('calendar', $prefs);

    echo $this->calendar->generate();
